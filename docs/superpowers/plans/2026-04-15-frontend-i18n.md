# Frontend I18n Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add persistent Chinese/English language switching to the React frontend, defaulting to Chinese and localizing all current shared and page-level UI strings plus time/date formatting.

**Architecture:** Introduce a small typed i18n layer under `frontend/src/i18n/`, keep locale state in a React provider, and migrate display labels plus formatting helpers to depend on the active locale. Integrate the provider near the app root, expose a header switcher, then convert shared components and pages to read localized messages instead of hard-coded strings.

**Tech Stack:** React 19, TypeScript, Vite, React Router, TanStack Query, Vitest, Testing Library, jsdom

```tsx
// frontend/src/i18n/I18nProvider.tsx
import { createContext, useContext, useEffect, useMemo, useState } from 'react'
import { DEFAULT_LOCALE, LOCALE_STORAGE_KEY, SUPPORTED_LOCALES, messages, type Locale } from './messages'

function readStoredLocale(): Locale {
  const stored = window.localStorage.getItem(LOCALE_STORAGE_KEY)
  return SUPPORTED_LOCALES.includes(stored as Locale) ? (stored as Locale) : DEFAULT_LOCALE
}

const I18nContext = createContext<{
  locale: Locale
  setLocale: (locale: Locale) => void
  t: (typeof messages)[Locale]
} | null>(null)

export function I18nProvider({ children }: { children: React.ReactNode }) {
  const [locale, setLocale] = useState<Locale>(() => readStoredLocale())

  useEffect(() => {
    window.localStorage.setItem(LOCALE_STORAGE_KEY, locale)
    document.documentElement.lang = locale
  }, [locale])

  const value = useMemo(() => ({ locale, setLocale, t: messages[locale] }), [locale])
  return <I18nContext.Provider value={value}>{children}</I18nContext.Provider>
}

export function useI18nContext() {
  const ctx = useContext(I18nContext)
  if (!ctx) throw new Error('useI18n must be used within I18nProvider')
  return ctx
}
```

```ts
export function formatDurationForLocale(seconds: number | undefined, locale: Locale): string {
  if (seconds == null) return '—'
  const h = Math.floor(seconds / 3600)
  const m = Math.floor((seconds % 3600) / 60)
  const s = Math.floor(seconds % 60)
  if (locale === 'zh-CN') {
    if (h > 0) return `${h}小时 ${m}分 ${s}秒`
    if (m > 0) return `${m}分 ${s}秒`
    return `${s}秒`
  }
  if (h > 0) return `${h}h ${m}m ${s}s`
  if (m > 0) return `${m}m ${s}s`
  return `${s}s`
}
```

```ts
// frontend/src/i18n/formatters.ts
import { messages, type Locale } from './messages'

export function formatRelativeTimeForLocale(dateStr: string, locale: Locale, now = new Date()): string {
  const date = new Date(dateStr)
  const diffSec = Math.floor((now.getTime() - date.getTime()) / 1000)
  if (diffSec < 60) return locale === 'zh-CN' ? '刚刚' : 'just now'
  const diffMin = Math.floor(diffSec / 60)
  if (diffMin < 60) return locale === 'zh-CN' ? `${diffMin}分钟前` : `${diffMin} minutes ago`
  const diffHour = Math.floor(diffMin / 60)
  if (diffHour < 24) return locale === 'zh-CN' ? `${diffHour}小时前` : `${diffHour} hours ago`
  const diffDay = Math.floor(diffHour / 24)
  if (diffDay === 1) return locale === 'zh-CN' ? '昨天' : 'yesterday'
  return date.toLocaleDateString(locale, { month: 'short', day: 'numeric' })
}

export function getStatusLabel(status: keyof typeof messages['zh-CN']['status'], locale: Locale) {
  return messages[locale].status[status]
}

export function getStageLabel(stage: keyof typeof messages['zh-CN']['stages'], locale: Locale) {
  return messages[locale].stages[stage]
}
```
